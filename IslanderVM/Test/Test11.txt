; Procs

STRUCTURE sum
    FIELD 4 : n
    FIELD 4 : acc
END


PROC calculateSum(sum& : s):
    STORE1 0
    LEA S3, s.n
    MOV S4, &S3
    CMP S4, S1
    JUMPEQ basecase
    LEA S2, s.acc
    ADD &S2, S4
    DEC &S3
    CALL calculateSum(s)
    basecase:
    STORE1 1 
    RETURN
